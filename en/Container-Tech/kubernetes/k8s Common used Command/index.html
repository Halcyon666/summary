K8S常用命令

状态检查​

# 检查 docker 状态
systemctl status docker.service
# 检查 kubelet 状态
systemctl status kubelet.service
# 检查 kubelet 端口
netstat -antp| grep 6443
netstat -pnlt | grep 6443
# 查看系统日志
tail -200f /var/log/messages
#  
kubectl get cs

重启命令​

systemctl restart docker.service
systemctl restart kubelet.service
